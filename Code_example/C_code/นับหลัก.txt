/*
 เช็คว่า input เป็นเลข 6 หหลักหรือไม่
 หลักการ:
      99999/100000    Min = 0.99999
    100000/100000   = 1.0
    999999/100000    Max = 9.99999
 */

#include <stdio.h>
int main() {
    long long int n;
    double x;
  
  do {
    printf("Number : ");
    scanf("%lld", &n);
    x = ((double)n) / 100000;

  } while ((x > 9.999999) || (x < 1.0));
}
