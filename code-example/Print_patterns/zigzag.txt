#include <iostream>
using namespace std;

int main()
{
    int n = 5;
    for(int i = 0; i < n; i++) {
        int m = (n - 1);
        for(int j = n; j > i; j--) {
            i % 2 == 0 ? cout << (n - m) : cout << (j - i);
            m--;
        }
        cout << endl;
    }
}
